{{/* Универсальный с AVIF + WebP + одним JPEG-fallback и LQIP-placeholder. Параметры через dict: src — Image Resource (.Resources.GetMatch / resources.Get) — обязательно alt — alt-текст (обязательно для значимых картинок, "" для декоративных) class — CSS-класс для sizes — sizes attribute (default "100vw") widths — slice ширин для srcset (default 480, 1080, 1920, 2560) loading — "lazy" | "eager" (default "lazy") fetchpriority — "high" | "low" (опционально — для hero ставить "high") Стратегия: 1. AVIF (q65) для современных браузеров (~90%) 2. WebP (q78) для широкой совместимости (~98%) 3. один JPEG (q80) в как fallback для IE/Safari ≤13 4. LQIP 20px blur-preview как inline base64 в CSS-variable --lqip на */}} {{ $src := .src }} {{ $alt := .alt | default "" }} {{ $class := .class | default "" }} {{ $sizes := .sizes | default "100vw" }} {{ $widths := .widths | default (slice 480 1080 1920 2560) }} {{ $loading := .loading | default "lazy" }} {{ $fp := .fetchpriority | default "" }} {{ if not $src }}{{ errorf "partial responsive-image: missing src (Image Resource required)" }}{{ end }} {{/* Не апскейлим — режем widths по исходной ширине */}} {{ $maxW := $src.Width }} {{ $usableWidths := slice }} {{ range $widths }} {{ $w := int . }} {{ if le $w $maxW }}{{ $usableWidths = $usableWidths | append $w }}{{ end }} {{ end }} {{ if not $usableWidths }}{{ $usableWidths = slice $maxW }}{{ end }} {{/* WebP srcset */}} {{ $webpSrcset := slice }} {{ range $usableWidths }} {{ $v := $src.Resize (printf "%dx webp q78" .) }} {{ $webpSrcset = $webpSrcset | append (printf "%s %dw" $v.RelPermalink .) }} {{ end }} {{/* Один JPEG fallback (без srcset — для очень старых браузеров) */}} {{ $midIdx := div (len $usableWidths) 2 }} {{ $midW := index $usableWidths $midIdx }} {{ $fallback := $src.Resize (printf "%dx jpg q80" $midW) }} {{/* LQIP — 20px blur preview как inline base64 в CSS-variable */}} {{ $lqip := $src.Resize "20x jpg q40" }} {{ $lqipURL := printf "data:image/jpeg;base64,%s" ($lqip.Content | base64Encode) }}
